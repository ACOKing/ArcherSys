# A demonstration of chaos

from turtle import *

N = 80

    return 3.9*x*(1-x)

    return 3.9*(x-x**2)

    return 3.9*x-3.9*x*x



[docs]def jumpto(x, y):
    penup(); goto(x,y)



[docs]def line(x1, y1, x2, y2):
    jumpto(x1, y1)
    pendown()
    goto(x2, y2)



[docs]def coosys():
    line(-1, 0, N+1, 0)
    line(0, -0.1, 0, 1.1)



[docs]def plot(fun, start, colour):
    pencolor(colour)
    x = start
    jumpto(0, x)
    pendown()
    dot(5)
    for i in range(N):
        x=fun(x)
        goto(i+1,x)
        dot(5)



[docs]def main():
    reset()
    setworldcoordinates(-1.0,-0.1, N+1, 1.1)
    speed(0)
    hideturtle()
    coosys()
    plot(f, 0.35, "blue")
    plot(g, 0.35, "green")
    plot(h, 0.35, "red")
    # Now zoom in:
    for s in range(100):
        setworldcoordinates(0.5*s,-0.1, N+1, 1.1)
    return "Done!"
