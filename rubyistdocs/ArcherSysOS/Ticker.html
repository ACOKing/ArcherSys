def run
  last_price = nil
  loop do
    price = Price.fetch(@symbol)
    print "Current price: #{price}\n"
    if price != last_price
      changed                 # notify observers
      last_price = price
      notify_observers(Time.now, price)
    end
    sleep 1
  end
end
