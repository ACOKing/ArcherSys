def build
  b = binding
  # create and run templates, filling member data variables
  ERB.new(<<-'END_PRODUCT'.gsub(/^\s+/, ""), 0, "", "@product").result b
    <%= PRODUCT[:name] %>
    <%= PRODUCT[:desc] %>
  END_PRODUCT
  ERB.new(<<-'END_PRICE'.gsub(/^\s+/, ""), 0, "", "@price").result b
    <%= PRODUCT[:name] %> -- <%= PRODUCT[:cost] %>
    <%= PRODUCT[:desc] %>
  END_PRICE
end
